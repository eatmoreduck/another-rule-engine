    // 比较运算符
    export type ComparisonOperator =
      | 'GT'       // >
      | 'GTE'      // >=
      | 'LT'       // <
      | 'LTE'      // <=
      | 'EQ'       // ==
      | 'NEQ'      // !=
      | 'CONTAINS' // contains
      | 'IN';      // in []

    // 决策结果
    export type DecisionAction = 'PASS' | 'REJECT' | 'MANUAL_REVIEW';

    // 单个条件
    export interface Condition {
      id: string;
      fieldName: string;       // 如 'amount', 'userLevel', 'ipAddress'
      operator: ComparisonOperator;
      value: string;           // 阈值
      valueType: 'NUMBER' | 'STRING' | 'BOOLEAN';
    }

    // 单个动作
    export interface Action {
      id: string;
      decision: DecisionAction;
      reason: string;          // 决策原因描述
    }

    // 条件-动作规则（对应一个 if-else 分支）
    export interface ConditionActionRule {
      conditions: Condition[];          // 条件列表（AND 关系）
      action: Action;                   // 满足条件时的动作
    }

    // 完整的表单规则配置
    export interface FormRuleConfig {
      defaultAction: Action;                  // 不满足任何条件时的默认动作
      rules: ConditionActionRule[];           // 条件-动作规则列表
    }

    // 预定义的特征字段（供用户选择）
    export interface FieldDefinition {
      name: string;         // 字段名
      label: string;        // 显示名
      valueType: 'NUMBER' | 'STRING' | 'BOOLEAN';
      description: string;  // 字段说明
    }

    // 可选的运算符列表
    export const OPERATOR_OPTIONS = [
      { value: 'GT', label: '大于 (>)' },
      { value: 'GTE', label: '大于等于 (>=)' },
      { value: 'LT', label: '小于 (<)' },
      { value: 'LTE', label: '小于等于 (<=)' },
      { value: 'EQ', label: '等于 (==)' },
      { value: 'NEQ', label: '不等于 (!=)' },
      { value: 'CONTAINS', label: '包含 (contains)' },
      { value: 'IN', label: '属于 (in [])' },
    ] as const;

    // 预定义特征字段列表
    export const PREDEFINED_FIELDS: FieldDefinition[] = [

      { name: 'orderCount', label: '订单数量', valueType: 'NUMBER', description: '用户历史订单数' },

      { name: 'isBlacklisted', label: '是否黑名单', valueType: 'BOOLEAN', description: '用户是否在黑名单中' },
