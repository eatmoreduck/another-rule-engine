    2. 创建 src/types/flowConfig.ts：

    ```typescript
    import type { Node, Edge } from '@xyflow/react'; // 或 'reactflow'

    // 节点类型枚举
    export enum FlowNodeType {
      START = 'start',
      END = 'end',
      CONDITION = 'condition',
      ACTION = 'action',
    }

    // 条件节点数据
    export interface ConditionNodeData {
      label: string;
      fieldName: string;         // 特征字段名
      operator: string;          // 比较运算符
      value: string;             // 阈值
      valueType: 'NUMBER' | 'STRING' | 'BOOLEAN';
      trueLabel?: string;        // "是" 分支标签
      falseLabel?: string;       // "否" 分支标签
    }

    // 动作节点数据
    export interface ActionNodeData {
      label: string;
      decision: 'PASS' | 'REJECT' | 'MANUAL_REVIEW';
      reason: string;
    }

    // 开始节点数据
    export interface StartNodeData {
      label: string;
    }

    // 结束节点数据
    export interface EndNodeData {
      label: string;
    }

    // 类型化的节点
    export type FlowNode = Node<ConditionNodeData | ActionNodeData | StartNodeData | EndNodeData>;

    // 流程图配置（用于持久化）
    export interface FlowConfig {
      nodes: FlowNode[];
      edges: Edge[];
    }
    ```

    3. 创建 src/utils/flowDslGenerator.ts：
       - 核心函数：`generateGroovyFromFlow(nodes: FlowNode[], edges: Edge[]): string`
       - 算法：
         a) 从 START 节点开始，沿 edges 进行图的拓扑遍历
         b) 遇到 CONDITION 节点：生成 `if (features.fieldName operator value) { ... } else { ... }`
         c) 遇到 ACTION 节点：生成 `return [decision: '...', reason: '...']`
         d) 遇到 END 节点：结束
       - 输出格式：
         ```groovy
         def evaluate(Map features) {
           if (features.amount > 1000) {
             return [decision: 'REJECT', reason: '金额超过阈值']
           }
           if (features.riskScore >= 80) {
             return [decision: 'MANUAL_REVIEW', reason: '高风险评分']
           }
           return [decision: 'PASS', reason: '默认通过']
         }
         ```
       - 处理分支：CONDITION 节点有两条出边（true/false），递归生成对应分支的代码
       - 需要处理循环检测（防止无限循环）
       - 辅助函数：
         - `getOutgoingEdges(nodeId, edges)` -- 获取出边
         - `getSourceHandle(edge)` -- 获取边的源 handle（用于判断 true/false 分支）
         - `findNodeById(nodeId, nodes)` -- 按 ID 查找节点
  </action>
