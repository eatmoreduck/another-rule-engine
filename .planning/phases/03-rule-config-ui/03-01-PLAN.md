GET    /api/v1/rules              -> Page<Rule> (page, size params)
GET    /api/v1/rules/{ruleKey}    -> Rule
POST   /api/v1/rules              -> Rule (body: CreateRuleRequest, header: X-Operator)
PUT    /api/v1/rules/{ruleKey}    -> Rule (body: UpdateRuleRequest, header: X-Operator)
DELETE /api/v1/rules/{ruleKey}    -> void (header: X-Operator)
POST   /api/v1/rules/{ruleKey}/enable  -> Rule (header: X-Operator)
POST   /api/v1/rules/{ruleKey}/disable -> Rule (header: X-Operator)
POST   /api/v1/rules/query        -> Page<Rule> (body: RuleQuery, page, size params)
POST   /api/v1/rules/validate     -> ValidateScriptResponse (body: ValidateScriptRequest)

<!-- 后端 DTO -->
CreateRuleRequest: { ruleKey: String, ruleName: String, ruleDescription?: String, groovyScript: String }
UpdateRuleRequest: { ruleName?: String, ruleDescription?: String, groovyScript?: String, changeReason?: String }
RuleQuery: { status?: RuleStatus, createdBy?: String, enabled?: Boolean, keyword?: String, createdAtStart?: LocalDateTime, createdAtEnd?: LocalDateTime }

       ```typescript
       export enum RuleStatus {
         DRAFT = 'DRAFT',
         ACTIVE = 'ACTIVE',
         ARCHIVED = 'ARCHIVED',
         DELETED = 'DELETED',
       }

       export interface Rule {
         id: number;
         ruleKey: string;
         ruleName: string;
         ruleDescription: string | null;
         groovyScript: string;
         version: number;
         status: RuleStatus;
         createdBy: string;
         createdAt: string;
         updatedBy: string | null;
         updatedAt: string | null;
         enabled: boolean;
         optLockVersion: number;
       }

       export interface PageResponse<T> {
         content: T[];
         totalElements: number;
         totalPages: number;
         number: number;
         size: number;
       }

       export interface CreateRuleRequest {
         ruleKey: string;
         ruleName: string;
         ruleDescription?: string;
         groovyScript: string;
       }

       export interface UpdateRuleRequest {
         ruleName?: string;
         ruleDescription?: string;
         groovyScript?: string;
         changeReason?: string;
       }

       export interface RuleQueryParams {
         status?: RuleStatus;
         keyword?: string;
         enabled?: boolean;
         page?: number;
         size?: number;
       }
       ```
    7. 创建 src/api/client.ts：配置 axios 实例，baseURL 为空（走 Vite proxy），设置默认 headers
    8. 创建 src/api/rules.ts：封装所有规则 API 调用函数

       - validateScript(groovyScript: string): Promise<ValidateScriptResponse>
    9. 创建 src/stores/ruleStore.ts：Zustand store 管理规则列表状态
