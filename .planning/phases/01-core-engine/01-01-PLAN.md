    **src/main/resources/application.yml:**
    ```yaml
    # Source: RESEARCH.md 模式5: 虚拟线程配置 + CONTEXT.md 决策 D-03
    spring:
      application:
        name: rule-engine

      # D-03: 启用虚拟线程（Java 21 + Spring Boot 3.3）
      threads:
        virtual:
          enabled: true

      # 缓存配置（D-13: Caffeine）
      cache:
        type: caffeine
        caffeine:
          spec: maximumSize=1000,expireAfterWrite=24h

      # Actuator 配置
      management:
        endpoints:
          web:
            exposure:
              include: health,info,metrics,prometheus
        metrics:
          export:
            prometheus:
              enabled: true
          tags:
            application: ${spring.application.name}

    # 服务器配置
    server:
      port: 8080
      tomcat:
        threads:
          max: 200  # 虚拟线程下可以设置更高的最大线程数
        accept-count: 100

    # 日志配置
    logging:
      level:
        root: INFO
        com.example.ruleengine: DEBUG
      pattern:
        console: "%d{yyyy-MM-dd HH:mm:ss} - %logger{36} - %msg%n"
    ```

    **src/test/resources/application-test.yml:**

    spring:
      application:
